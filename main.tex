\section{Bost-Connes Criterion}

\begin{definition}[Bost Sum]
$$C(\alpha) := \sum_{p\in S(\alpha)} \frac{\log p}{p-1}.$$
\end{definition}

\begin{lemma}[Bost-Connes for Weight 3]
\label{lem:bost}
If $C(\alpha) > 2\sqrt{g(X_0(N))}$, then $L(f,s)$ satisfies the Generalized Riemann Hypothesis.
\end{lemma}

For $\alpha = \pi/10$, we compute:
\begin{equation}
C\left(\frac{\pi}{10}\right) = \sum_{p\in S_{14}} \frac{\log p}{p-1} = 1.43456736419097\ldots > 0 = 2\sqrt{0}.
\end{equation}

Since $g(X_0(10)) = 0$, the criterion is satisfied.
